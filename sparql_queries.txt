6)
SELECT ?director (COUNT(*) AS ?moviecount)
 WHERE {
?x1 foaf:firstName "Harvey"^^rdfs:Literal;
      foaf:lastName "Keitel"^^rdfs:Literal.
  ?director foaf:made ?movie.
  ?x1 foaf:acts ?movie.
}
GROUP BY (?director)
ORDER BY DESC(?moviecount)
LIMIT 1

8)
SELECT distinct ?dir
 WHERE {
?x1 foaf:firstName "Harvey"^^rdfs:Literal;
      foaf:lastName "Keitel"^^rdfs:Literal.
?x2 foaf:firstName "John"^^rdfs:Literal;
      foaf:lastName "Travolta"^^rdfs:Literal.
  ?dir foaf:made ?mov.
   ?x1 foaf:acts ?mov.
   ?x2 foaf:acts ?mov.
   ?mov foaf:launchDate ?date.
  FILTER (?date > 1950 && ?date < 2000)
 }

9)
SELECT distinct ?actor
 WHERE {
?x1 foaf:firstName "Harvey"^^rdfs:Literal;
      foaf:lastName "Keitel"^^rdfs:Literal.
?x2 foaf:firstName "John"^^rdfs:Literal;
      foaf:lastName "Travolta"^^rdfs:Literal.
    ?dir foaf:made ?mov.
    ?x1 foaf:acts ?mov.
    ?x2 foaf:acts ?mov.
    ?actor foaf:acts ?mov.
    ?mov foaf:launchDate ?date.
  FILTER (?date > 1950 && ?date < 2000 && ?actor != ?x1 && ?actor != ?x2)
 }
